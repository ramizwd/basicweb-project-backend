CREATE TABLE `user` (
  `id` int PRIMARY KEY NOT NULL,
  `username` varchar(255),
  `email` varchar(255),
  `password` varchar(255),
  `description` text,
  `role` int(1) NOT NULL DEFAULT '1'
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `user_post` (
  `id` int PRIMARY KEY NOT NULL,
  `date` date,
  `title` text(100),
  `filename` text,
  `text` text,
  `vote` int,
  `user_id` int
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `comments` (
  `id` int PRIMARY KEY NOT NULL,
  `comment` text,
  `vote` int,
  `user_post_id` int,
  `user_id` int
)ENGINE=MyISAM DEFAULT CHARSET=utf8;
