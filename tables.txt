CREATE TABLE `pjr_user` (
  `user_id` int PRIMARY KEY NOT NULL AUTO_INCREMENT,
  `username` varchar(100) NOT NULL,
  `profile_picture` text,
  `email` varchar(255) NOT NULL,
  `password` varchar(255) NOT NULL,
  `description` text,
  `role` int(1) NOT NULL DEFAULT "1"
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `pjr_post` (
  `post_id` int PRIMARY KEY NOT NULL AUTO_INCREMENT,
  `date` date NOT NULL,
  `title` text(100) NOT NULL,
  `filename` text,
  `description` text,
  `poster` int NOT NULL
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `pjr_comments` (
  `comments_id` int PRIMARY KEY NOT NULL AUTO_INCREMENT,
  `comment` text NOT NULL,
  `post_id` int NOT NULL,
  `user_id` int NOT NULL
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `pjr_vote` (
  `user_id` int NOT NULL,
  `vote` int NOT NULL,
  `vote_type` text NOT NULL,
  `post_id` int NOT NULL
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

CREATE TABLE `pjr_comments_vote` (
  `user_id` int NOT NULL,
  `vote` int NOT NULL,
  `comments_id` int NOT NULL
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

ALTER TABLE `pjr_user` ADD FOREIGN KEY (`user_id`) REFERENCES `pjr_post` (`poster`);

ALTER TABLE `pjr_post` ADD FOREIGN KEY (`post_id`) REFERENCES `pjr_comments` (`post_id`);

ALTER TABLE `pjr_user` ADD FOREIGN KEY (`user_id`) REFERENCES `pjr_comments` (`user_id`);

ALTER TABLE `pjr_user` ADD FOREIGN KEY (`user_id`) REFERENCES `pjr_vote` (`user_id`);

ALTER TABLE `pjr_post` ADD FOREIGN KEY (`post_id`) REFERENCES `pjr_vote` (`post_id`);

ALTER TABLE `pjr_comments` ADD FOREIGN KEY (`comments_id`) REFERENCES `pjr_comments_vote` (`comments_id`);

ALTER TABLE `pjr_user` ADD FOREIGN KEY (`user_id`) REFERENCES `pjr_comments_vote` (`user_id`);
